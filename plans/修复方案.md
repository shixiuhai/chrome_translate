# Chrome翻译插件修复方案

## 问题总结

### 1. 全文翻译后页面未显示目标语言
**根本原因：**
- [content.js:213-216](content.js:213) 中弹窗使用了`position: fixed`但定位值包含`scrollY/scrollX`，导致弹窗位置计算错误
- [content.js:102](content.js:102) 仅修改`textContent`，不支持HTML格式的翻译结果
- 节点过滤规则过于严格，跳过了很多需要翻译的元素
- 错误处理不完善，翻译失败时没有回退机制
- 动态加载的内容无法被翻译
- 翻译后的文本没有添加样式标记，无法与原文本区分

### 2. 选中翻译后弹窗不出现
**根本原因：**
- [content.js:214-216](content.js:214) 弹窗定位逻辑错误，fixed定位使用了滚动偏移量导致弹窗显示在视口外
- [content.js:320-324](content.js:320) 点击外部关闭事件在弹窗显示前就触发，导致弹窗刚创建就被移除
- 选中文本后页面滚动导致位置计算错误
- 与页面其他元素的z-index冲突
- 通知和弹窗的z-index相同导致覆盖
- 暗色主题适配逻辑错误，导致弹窗背景和文字颜色相同

## 修复方案

### 一、全文翻译功能修复

#### 1. 优化文本节点收集逻辑
```javascript
// 改进节点过滤器，增加更多可翻译元素类型
acceptNode: function(node) {
  const parent = node.parentElement;
  if (!parent) return NodeFilter.FILTER_REJECT;
  
  const tagName = parent.tagName.toUpperCase();
  const excludedTags = ['SCRIPT', 'STYLE', 'NOSCRIPT', 'IFRAME', 'INPUT', 'TEXTAREA', 'SELECT', 'CODE', 'PRE'];
  
  if (
    excludedTags.includes(tagName) ||
    parent.isContentEditable ||
    node.textContent.trim().length < 2 || // 保留短文本翻译
    /^\d+([,.]\d+)*$/.test(node.textContent.trim()) // 改进纯数字判断
  ) {
    return NodeFilter.FILTER_REJECT;
  }
  
  // 特殊处理按钮元素，仅翻译文本内容
  if (tagName === 'BUTTON' && node.textContent.trim().length > 0) {
    return NodeFilter.FILTER_ACCEPT;
  }
  
  return NodeFilter.FILTER_ACCEPT;
}
```

#### 2. 支持HTML格式翻译结果
```javascript
// 改进翻译结果应用逻辑
batch.forEach((item, index) => {
  if (!originalTexts.has(item.node)) {
    originalTexts.set(item.node, {
      text: item.text,
      isHTML: false
    });
  }
  
  if (translations[index]) {
    const originalNodeText = item.node.textContent;
    const leadingWhitespace = originalNodeText.match(/^\s*/)[0];
    const trailingWhitespace = originalNodeText.match(/\s*$/)[0];
    
    // 检查翻译结果是否包含HTML标签
    if (/<[a-z][\s\S]*>/i.test(translations[index])) {
      // 使用innerHTML插入HTML内容
      const parent = item.node.parentElement;
      if (parent && !parent.querySelector('.translated-text')) {
        const span = document.createElement('span');
        span.className = 'translated-text';
        span.dataset.original = item.text;
        span.innerHTML = translations[index];
        parent.replaceChild(span, item.node);
        originalTexts.set(span, {
          text: item.text,
          isHTML: true
        });
      }
    } else {
      // 普通文本替换
      item.node.textContent = leadingWhitespace + translations[index] + trailingWhitespace;
    }
  }
});
```

#### 3. 改进错误处理和重试机制
```javascript
// 增加失败重试逻辑
const MAX_RETRIES = 2;
for (let i = 0; i < textNodes.length; i += batchSize) {
  const batch = textNodes.slice(i, i + batchSize);
  const texts = batch.map(item => item.text);
  
  let retries = 0;
  let success = false;
  
  while (retries < MAX_RETRIES && !success) {
    try {
      const response = await chrome.runtime.sendMessage({
        action: 'translate',
        data: {
          q: texts,
          source,
          target,
          format: 'html'
        }
      });

      if (response.success) {
        // 处理翻译结果
        success = true;
      } else {
        throw new Error(response.error);
      }
    } catch (error) {
      retries++;
      if (retries === MAX_RETRIES) {
        console.error('批次翻译失败，已重试' + MAX_RETRIES + '次:', error);
        showNotification(`部分内容翻译失败: ${error.message}`, 'warning');
      } else {
        // 等待后重试
        await new Promise(resolve => setTimeout(resolve, 1000 * retries));
      }
    }
  }
}
```

#### 4. 添加动态内容监听
```javascript
// 监听DOM变化，翻译动态加载的内容
function observeDynamicContent() {
  const observer = new MutationObserver((mutations) => {
    if (isTranslating || originalTexts.size === 0) return;
    
    mutations.forEach(mutation => {
      if (mutation.addedNodes.length > 0) {
        mutation.addedNodes.forEach(node => {
          if (node.nodeType === Node.TEXT_NODE && node.textContent.trim().length > 1) {
            // 对新添加的文本节点进行翻译
            translateSingleNode(node, currentTranslationTarget);
          } else if (node.nodeType === Node.ELEMENT_NODE) {
            // 递归处理元素节点内的文本
            const textNodes = [];
            const walker = document.createTreeWalker(
              node,
              NodeFilter.SHOW_TEXT,
              { acceptNode: () => NodeFilter.FILTER_ACCEPT }
            );
            
            let textNode;
            while (textNode = walker.nextNode()) {
              if (textNode.textContent.trim().length > 1) {
                textNodes.push(textNode);
              }
            }
            
            if (textNodes.length > 0) {
              // 批量翻译新节点
              translateBatch(textNodes, currentTranslationTarget);
            }
          }
        });
      }
    });
  });
  
  observer.observe(document.body, {
    childList: true,
    subtree: true
  });
  
  return observer;
}
```

### 三、其他优化

#### 1. 改进还原功能
```javascript
function restoreOriginalPage() {
  if (isTranslating) {
    showNotification('正在翻译中，无法还原');
    return;
  }

  if (originalTexts.size === 0) {
    showNotification('没有可还原的内容');
    return;
  }

  try {
    originalTexts.forEach((data, node) => {
      if (node.parentNode) {
        if (data.isHTML && node.classList.contains('translated-text')) {
          // 还原HTML元素为原始文本节点
          const textNode = document.createTextNode(data.text);
          node.parentNode.replaceChild(textNode, node);
        } else {
          // 还原普通文本节点
          node.textContent = data.text;
        }
      }
    });
    originalTexts.clear();
    showNotification('页面已还原到原始状态');
  } catch (error) {
    showNotification(`还原失败: ${error.message}`, 'error');
  }
}
```
